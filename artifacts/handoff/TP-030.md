### 3. WebSocket 데이터 통합 — useCoworkState 훅

#### dashboard/hooks/useCoworkState.ts (신규, ~80줄)

단일 WebSocket 연결로 모든 cowork 데이터를 관리:

```typescript
export function useCoworkState() {
  const [agents, setAgents] = useState<AgentState[]>([]);
  const [activeWork, setActiveWork] = useState<WorkItem[]>([]);
  const [odinMessages, setOdinMessages] = useState<OdinMessage[]>([]);
  const [plan, setPlan] = useState<PlanState | null>(null);

  useEffect(() => {
    const wsStatus = new WebSocket(`ws://${location.host}/ws/status`);
    const wsOdin = new WebSocket(`ws://${location.host}/ws/odin`);

    wsStatus.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      switch (msg.type) {
        case "agent_status": setAgents(...); break;
        case "agent_progress": updateActiveWork(...); break;
        case "plan_progress": setPlan(...); break;
        case "context_shared": /* notification */ break;
      }
    };

    wsOdin.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      setOdinMessages(prev => [...prev.slice(-50), msg]);
    };

    return () => { wsStatus.close(); wsOdin.close(); };
  }, []);

  return { agents, activeWork, odinMessages, plan };
}
```
